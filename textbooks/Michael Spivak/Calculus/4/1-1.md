(i) 
$$
\begin{aligned}
ax = a & \implies a^{-1} a x = a^{-1} a && a \neq 0 \\
& \implies x = 1
\end{aligned}
$$


(ii) 
$$
\begin{aligned}
x^{2} - y^{2} &= x^{2} - y^{2} + 0 \\
&= x^{2} - y^{2} + xy - xy \\
&= x (x + y) - y (x + y) \\
&= (x - y)(x + y)
\end{aligned}
$$


(iii) 
$$
\begin{aligned}
x^{2} = y^{2} & \implies x^{2} - y^{2} = 0 \\
& \implies (x - y)(x + y) = 0 \\
& \implies x - y = 0 \; \vee \; x + y = 0 \\
& \implies x = y \; \vee \; x = -y
\end{aligned}
$$
